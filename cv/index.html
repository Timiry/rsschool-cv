using System;
using System.Text;

public class Kata
{
    public static int SquareDigits(int n)
    {
    if(n == 0) return 0;
    StringBuilder buf = new StringBuilder();
    int a;
    while (n > 0)
    {
        a = n % 10;
        a = a*a;
        n /= 10;
        buf.Insert(0, a.ToString());
    }
    return Convert.ToInt32(buf.ToString());
    }
}
